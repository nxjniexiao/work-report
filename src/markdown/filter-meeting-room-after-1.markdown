变化的条件：<br>
1： min -- 最小值<br>
2： max -- 最大值<br>

```js
$scope.galleryfulList = [
  { id: 0, name: "全部" },
  { id: 1, name: "0-10", min: 0, max: 10 },
  { id: 2, name: "11-20", min: 11, max: 20 },
  { id: 3, name: "21-30", min: 21, max: 30 },
  { id: 4, name: "31-40", min: 31, max: 40 },
  { id: 5, name: "40人以上", min: 41 }
];
```
